<div className="">
          <label className="form-label fw-bold mb-3">Select Quantity:</label>
             <div className="d-flex justify-content-start gap-2">
                  <button className="btn quantity-btn" onClick={decrement}>
                      <i className="fas fa-minus"></i>
                  </button>
                      <input type="number" className="form-control quantity-input" id="quantityInput"
                            value={qty} min="1" max="99" style={{ width: '60px' }} readOnly />
                   <button className="btn quantity-btn" onClick={increment}>
                        <i className="fas fa-plus"></i>
                   </button>
            </div>
</div>


const [qty, setQty] = useState(1);

const decrement = () => {
    if (qty > 1) {
     setQty((q) => q - 1)
     }
}
const increment = () => {
    setQty((q) => q + 1)
}
